39 void polyfit(const std::vector<double> &xv, const std::vector<double> &yv, std::vector<double> &coeff, int order) {
40  Eigen::MatrixXd A(xv.size(), order+1);
41  Eigen::VectorXd yv_mapped = Eigen::VectorXd::Map(&yv.front(), yv.size());
42  Eigen::VectorXd result;
43 
44  // create matrix
45  for (size_t i = 0; i < xv.size(); i++)
46  for (size_t j = 0; j < order+1; j++)
47  A(i, j) = pow(xv.at(i), j);
48 
49  // solve for linear least squares fit
50  result = A.householderQr().solve(yv_mapped);
51 
52  coeff.resize(order+1);
53  for (size_t i = 0; i < order+1; i++)
54  coeff[i] = result[i];
55 }
